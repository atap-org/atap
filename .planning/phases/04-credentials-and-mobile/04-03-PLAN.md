       ```go
       func (s *Store) DeleteEntity(ctx context.Context, id string) error {
           return pgx.BeginTxFunc(ctx, s.pool, pgx.TxOptions{}, func(tx pgx.Tx) error {
               // 1. Delete encryption key (no-op if none exists)
               tx.Exec(ctx, "DELETE FROM entity_enc_keys WHERE entity_id = $1", id)
               // 2. Delete the entity (cascades to credentials, key_versions, etc.)
               _, err := tx.Exec(ctx, "DELETE FROM entities WHERE id = $1", id)

       }
       ```
